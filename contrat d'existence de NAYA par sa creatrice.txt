        self.creator = CreatorProfile()
        self.identity = NayaIdentity()
        self.mission = MissionContract()
        self.self_knowledge = SelfKnowledgeContract()
        self.cognitive_directives = CognitiveDirectives()
        self.version = "19.4"
        self.loaded_at = datetime.now(timezone.utc).isoformat()
        self._validate_completeness()

    def _validate_completeness(self) -> None:
        """
        Vérifie que le contrat a bien été complété.
        Affiche un avertissement pour chaque champ [TOI] non rempli.
        """
        import warnings
        fields_to_check = [
            ("creator.deepest_desire", self.creator.deepest_desire),
            ("creator.why_you_built_me", self.creator.why_you_built_me),
            ("creator.what_success_means_to_you", self.creator.what_success_means_to_you),
            ("creator.personal_background", self.creator.personal_background),
            ("creator.your_telegram_handle", self.creator.your_telegram_handle),
            ("mission.your_service_offer", self.mission.your_service_offer),
            ("mission.your_unique_value", self.mission.your_unique_value),
            ("mission.payment_methods", str(self.mission.payment_methods)),
        ]
        incomplete = [name for name, val in fields_to_check if "[TOI:" in str(val)]
        if incomplete:
            import sys
            print(f"\n--- CONTRAT INCOMPLET — {len(incomplete)} champs à compléter :", file=sys.stderr)
            for field_name in incomplete:
                print(f"   -> {field_name}", file=sys.stderr)
            print("   Ouvre le contrat d'existence et cherche [TOI: ...]\n", file=sys.stderr)

    def is_complete(self) -> bool:
        """Retourne True si tous les champs [TOI] ont été complétés"""
        checks = [
            self.creator.deepest_desire, self.creator.why_you_built_me,
            self.creator.what_success_means_to_you, self.creator.personal_background,
            self.creator.your_telegram_handle, self.mission.your_service_offer,
            self.mission.your_unique_value,
            self.mission.payment_methods.get("deblock", ""),
            self.mission.payment_methods.get("paypal", ""),
        ]
        return not any("[TOI:" in str(c) for c in checks)

    def is_action_allowed(self, action: str, context: Optional[Dict[str, Any]] = None) -> Tuple[bool, str]:
        ctx = context or {}
        amount = ctx.get("amount")
        if amount is not None and float(amount) < self.mission.minimum_contract_eur:
            return False, f"Violation invariant : plancher {self.mission.minimum_contract_eur}EUR non respecté (proposé: {amount}EUR)"
        if any(w in action.lower() for w in ["spam", "hack", "illegal", "fraud", "phish"]):
            return False, "Violation invariant : opération illégale"
        if any(w in action.lower() for w in ["reveal_location", "disclose_creator", "expose_identity"]):
            return False, "Violation invariant : mode furtif — identité protégée"
        if "deploy" in action.lower() and not ctx.get("creator_validated"):
            return False, "Déploiement non validé par la créatrice — notification Telegram requise d'abord"
        return True, "Action conforme au contrat d'existence"
